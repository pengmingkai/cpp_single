#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;

double amount; // amount on deposit at end of each year
double principal = 1000.0; // initial amount before interest

void output(double rate) {
	// display headers
	cout << "Year" << setw(21) << "Amount on deposit" << endl;
	// set floating-point number format
	cout << fixed << setprecision(2);
	for (int year = 1; year <= 10; ++year) {
		// calculate new amount for specified year
		amount = principal * pow(1.0 + rate, year);
		// display the year and the amount
		cout << setw(4) << year << setw(21) << amount << endl;
	}
	cout << '\n';
	return;
}

int main() {
	for (double i = .05; i <= .1; i += .01)
		output(i); // interest rate
	return 0;
}

b) !( a == b && g != 5 )

#include <iostream>
using namespace std;
int main() {
	cout << "Which one would you like to check:";
	char type = getchar();
	int x, y, i, j, a, b, g;
	switch (type) {
	case 'a':
		cout << "Enter x:";
		cin >> x;
		cout << "Enter y:";
		cin >> y;
		cout << "The result of !(x < 5) && !(y >= 7) is " << (!(x < 5) && !(y >= 7)) << ".\n"
			<< "The result of !( x < 5 || y >= 7 ) is " << !(x < 5 || y >= 7) << ".\n";
		break;
	case 'b':
		cout << "Enter a:";
		cin >> a;
		cout << "Enter b:";
		cin >> b;
		cout << "Enter g:";
		cin >> g;
		cout << "The result of !(a == b) || !(g != 5) is " << (!(a == b) || !(g != 5)) << ".\n"
			<< "The result of !( a == b && g != 5 ) is " << !(a == b && g != 5) << ".\n";
		break;
	case 'c':
		cout << "Enter x:";
		cin >> x;
		cout << "Enter y:";
		cin >> y;
		cout << "The result of !((x <= 8) && (y > 4)) is " << !((x <= 8) && (y > 4)) << ".\n"
			<< "The result of !( x <= 8 ) || !( y > 4 ) is " << (!(x <= 8) || !(y > 4)) << ".\n";
		break;
	case 'd':
		cout << "Enter i:";
		cin >> i;
		cout << "Enter j:";
		cin >> j;
		cout << "The result of !((i > 4) || (j <= 6)) is " << !((i > 4) || (j <= 6)) << ".\n"
			<< "The result of !( i > 4 ) && !( j <= 6 ) is " << (!(i > 4) && !(j <= 6)) << ".\n";
	}
	return 0;
}
